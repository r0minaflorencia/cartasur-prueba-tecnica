CREATE DATABASE cartasur;
USE cartasur;

CREATE TABLE clientes (
    id_cliente INT AUTO_INCREMENT PRIMARY KEY,
    dni VARCHAR(15) NOT NULL UNIQUE,
    nombre VARCHAR(100) NOT NULL,
    direccion_envio VARCHAR(150)
);

CREATE TABLE sucursales (
    id_sucursal INT AUTO_INCREMENT PRIMARY KEY,
    nombre_sucursal VARCHAR(100) NOT NULL,
    direccion_sucursal VARCHAR(150)
);

CREATE TABLE empleados (
    id_empleado INT AUTO_INCREMENT PRIMARY KEY,
    nombre_empleado VARCHAR(100) NOT NULL
);

CREATE TABLE productos (
    id_producto INT AUTO_INCREMENT PRIMARY KEY,
    nombre_producto VARCHAR(100) NOT NULL,
    precio_unitario DECIMAL(10,2) NOT NULL
);

CREATE TABLE ventas (
    id_venta INT AUTO_INCREMENT PRIMARY KEY,
    fecha_venta DATETIME DEFAULT CURRENT_TIMESTAMP,
    id_cliente INT,
    id_sucursal INT,
    id_empleado INT,
    importe_total DECIMAL(10,2) DEFAULT 0.00,
    FOREIGN KEY (id_cliente) REFERENCES clientes(id_cliente),
    FOREIGN KEY (id_sucursal) REFERENCES sucursales(id_sucursal),
    FOREIGN KEY (id_empleado) REFERENCES empleados(id_empleado)
);

    id_detalle INT AUTO_INCREMENT PRIMARY KEY,
    id_venta INT NOT NULL,

-- Desactivo el modo seguro para actualizar el importe total:
SET SQL_SAFE_UPDATES = 0;

UPDATE ventas v
SET v.importe_total = (
    SELECT SUM(d.cantidad * d.precio_unitario_en_momento_venta)
    FROM detalle_ventas d
    WHERE d.id_venta = v.id_venta
)
WHERE v.importe_total = 0;

SET SQL_SAFE_UPDATES = 1; -- Vuelvo al modo seguro.

-- Obtengo la fecha en la que hubo más ventas:
SELECT DATE(fecha_venta) AS fecha, 
COUNT(*) AS total_ventas FROM ventas
GROUP BY fecha
LIMIT 1;
